Description

Driver API for WDT(Watchdog Timer)

Example

Overview

The following example shows the usage of the WDT interface. In this example, we'll start to kick WDT for 5 times and then stop, the WDT is configured as WDT_InterruptResetMode and we won't clear WDT interrupt flag upon timeout, so a system reset shall follow a WDT interrupt, the time tick is achieved through TIMER's interrupt.

Hardware Requirements

  • EVK board
  • Mini USB cable
  • Personal Computer

Board Settings

This example project does not need any special hardware configurations. Just keep board's jumper settings and configurations in default state.

Prepare and Run Example

  1. Connect Mini USB cable between PC host and UART1 port.
  2. Open a serial ternimal with settings below:
    • 115200 baud rate
    • 8 data bits
    • No parity
    • 1 stop bit
    • No flow control
  3. Select exampleId to WDT_EX in app.c, compile and download the program to the board
  4. Press reset button on the board or lauch the debugger of IDE to begin running the example.
  5. When the example runs successfully, below information will be printed on the serial terminal.
    tick:1, wdt interrupt count: 0
    wdt kick: 1
    tick:2, wdt interrupt count: 0
    wdt kick: 2
    tick:3, wdt interrupt count: 0
    wdt kick: 3
    tick:4, wdt interrupt count: 0
    wdt kick: 4
    tick:5, wdt interrupt count: 0
    Last kick, wdt interrupt will occur in 5 ticks
    tick:6, wdt interrupt count: 0
    tick:7, wdt interrupt count: 0
    tick:8, wdt interrupt count: 0
    tick:9, wdt interrupt count: 0

#include <stdio.h>
#include <string.h>
#include "bsp.h"
#include "timer_ec626.h"
#include "wdt_ec626.h"
/*
This project demos the usage of WDT, in this example, we'll start to kick WDT for 5 times and then stop,
the WDT is configured as WDT_InterruptResetMode and we won't clear WDT interrupt flag upon timeout,
so a system reset shall follow a WDT interrupt, the time tick is achieved through TIMER's interrupt.
*/
/** \brief WDT counter refresh times */
#define WDT_KICK_TIMES (5)
/** \brief WDT interrupt count */
static volatile int wdtInterruptCount = 0;
/** \brief timer interupt count */
static volatile int timerInterruptCount = 0;
/** \brief Flag to control message print */
static volatile int printDone = 1;
/**
\fn void NMI_Handler()
\brief wdt interrupt is NMI
\return
*/
void NMI_Handler()
{
wdtInterruptCount++;
}
/**
\fn void Timer0_ISR()
\brief timer0 interrupt service routine
\return
*/
void Timer0_ISR()
{
timerInterruptCount++;
printDone = 0;
if(timerInterruptCount <= WDT_KICK_TIMES)
WDT_Kick();
}
void WDT_ExampleEntry(void)
{
// Config WDT clock, source from 40KHz and divide by 5
CLOCK_SetClockSrc(GPR_WDGFuncClk, GPR_WDGClkSel_40K);
CLOCK_SetClockDiv(GPR_WDGFuncClk, 5);
// WDT config, timeout is 5s
wdt_config_t wdtConfig;
wdtConfig.mode = WDT_InterruptResetMode;
wdtConfig.timeoutValue = 40000;
WDT_Init(&wdtConfig);
// Config TIMER clock, source from 40KHz and divide by 1
CLOCK_SetClockSrc(GPR_TIMER0FuncClk, GPR_TIMER0ClkSel_40K);
CLOCK_SetClockDiv(GPR_TIMER0FuncClk, 1);
TIMER_DriverInit();
// Config timer period as 1s, match0 value is 40000
timer_config_t timerConfig;
TIMER_GetDefaultConfig(&timerConfig);
timerConfig.initValue = 1;
timerConfig.reloadOption = TIMER_ReloadOnMatch0;
timerConfig.match0 = 40000;
TIMER_Init(0, &timerConfig);
// Note interrupt flag won't assert in TIMER_InterruptPulse mode
TIMER_InterruptConfig(0, TIMER_Match0Select, TIMER_InterruptPulse);
TIMER_InterruptConfig(0, TIMER_Match1Select, TIMER_InterruptDisabled);
TIMER_InterruptConfig(0, TIMER_Match2Select, TIMER_InterruptDisabled);
// Enable IRQ
XIC_SetVector(PXIC_Timer0_IRQn, Timer0_ISR);
XIC_EnableIRQ(PXIC_Timer0_IRQn);
WDT_Start();
TIMER_Start(0);
while(1)
{
if (printDone == 0)
{
printf("tick:%d, wdt interrupt count: %d\n", timerInterruptCount, wdtInterruptCount);
if (timerInterruptCount < WDT_KICK_TIMES)
printf("wdt kick: %d\n", timerInterruptCount);
else if (timerInterruptCount == WDT_KICK_TIMES)
printf("Last kick, wdt interrupt will occur in 5 ticks\n");
else if (timerInterruptCount == WDT_KICK_TIMES * 2)
printf("System will reset in 5 ticks\n");
printDone = 1;
}
}
}
