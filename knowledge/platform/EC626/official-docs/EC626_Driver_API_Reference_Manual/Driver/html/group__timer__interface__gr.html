Description

Driver API for TIMER.

Example

Overview

The following example shows the usage of the TIMER interface. In this example, we'll use 2 instance of TIMER, one is used to output pwm, and its frequency is configure as 1KHz while duty cycle is controlled by another TIMER, the duty cycle is updated in interval of 5 from 0 to 100 periodically, the period is achieved with aid of another instance of timer whose interrupt is triggered every 500ms.

Hardware Requirements

  • EVK board
  • Mini USB cable
  • Personal Computer

Board Settings

This example project does not need any special hardware configurations. Just keep board's jumper settings and configurations in default state.

Prepare and Run Example

  1. Connect Mini USB cable between PC host and UART1 port.
  2. Open a serial ternimal with settings below:
    • 115200 baud rate
    • 8 data bits
    • No parity
    • 1 stop bit
    • No flow control
  3. Select exampleId to TIMER_EX in app.c, compile and download the program to the board
  4. Press reset button on the board or lauch the debugger of IDE to begin running the example.
  5. When the example runs successfully, below information will be printed on the serial terminal and LED RED changes its birghtness.
    TIMER Example Start

* Description: EC626 TIMER driver example entry source file
* History: Rev1.0 2018-07-17

#include <stdio.h>
#include "timer_ec626.h"
#include "bsp.h"
#include "app.h"
#define TOKEN_CAT2(a, b) a##b
#define TOKEN_PASTE2(a, b) TOKEN_CAT2(a, b)
#define TOKEN_PASTE3(a, b, c) TOKEN_PASTE2(TOKEN_PASTE2(a, b), c)
/*
This project demos PWM/COUNTER/CAPTURE feature of TIMER
*/
// use button to test pwm stop option
#define BUTTON_GPIO_INSTANCE (0)
#define BUTTON_GPIO_PIN (0)
#define BUTTON_PAD_INDEX (5)
#define BUTTON_PAD_ALT_FUNC (PAD_MuxAlt0)
/** \brief TIMER instance for PWM output */
#define PWM_INSTANCE 5
/** \brief PWM clock id for clock configuration */
#define PWM_CLOCK_ID TOKEN_PASTE3(GPR_TIMER, PWM_INSTANCE, FuncClk)
/** \brief PWM clock source select */
#define PWM_CLOCK_SOURCE TOKEN_PASTE3(GPR_TIMER, PWM_INSTANCE, ClkSel_40K)
/** \brief PWM initial duty cycle percent */
#define PWM_INIT_DUTY_CYCLE_PERCENT (50)
/** \brief TIMER/PWM out pin */
#define PWM_OUT_PIN (11)
/** \brief TIMER/PWM Alt function select */
#define PWM_OUT_ALT_SEL (PAD_MuxAlt5)
/** \brief TIMER/PWM out pin */
#define PWM_OUT_N_PIN (10)
/** \brief TIMER/PWM Alt function select */
#define PWM_OUT_N_ALT_SEL (PAD_MuxAlt6)
/** \brief TIMER instance */
#define TIMER_INSTANCE 1
/** \brief TIMER clock id for clock configuration */
#define TIMER_CLOCK_ID TOKEN_PASTE3(GPR_TIMER, TIMER_INSTANCE, FuncClk)
/** \brief TIMER clock source select */
#define TIMER_CLOCK_SOURCE TOKEN_PASTE3(GPR_TIMER, TIMER_INSTANCE, ClkSel_40K)
/** \brief TIMER instance IRQ number */
#define TIMER_INSTANCE_IRQ TOKEN_PASTE3(PXIC_Timer, TIMER_INSTANCE, _IRQn)
/** \brief COUNTER/CAPTURE instance */
#define COUNTER_INSTANCE 4
#define CAPTURE_INSTANCE COUNTER_INSTANCE
/** \brief COUNTER/CAPTURE clock id for clock configuration */
#define COUNTER_CLOCK_ID TOKEN_PASTE3(GPR_TIMER, COUNTER_INSTANCE, FuncClk)
#define CAPTURE_CLOCK_ID COUNTER_CLOCK_ID
/** \brief COUNTER/CAPTURE clock source select */
#define COUNTER_CLOCK_SOURCE TOKEN_PASTE3(GPR_TIMER, COUNTER_INSTANCE, ClkSel_40K)
#define CAPTURE_CLOCK_SOURCE COUNTER_CLOCK_SOURCE
/** \brief COUNTER/CAPTURE instance IRQ number */
#define COUNTER_INSTANCE_IRQ TOKEN_PASTE3(PXIC_Timer, COUNTER_INSTANCE, _IRQn)
#define CAPTURE_INSTANCE_IRQ COUNTER_INSTANCE_IRQ
/** \brief COUNTER/CAPTURE input pin, GPIO2 */
#define COUNTER_INPUT_ALT_SEL (PAD_MuxAlt0)
#define COUNTER_INPUT_PIN (7)
#define CAPTURE_INPUT_ALT_SEL COUNTER_INPUT_ALT_SEL
#define CAPTURE_INPUT_PIN COUNTER_INPUT_PIN
volatile static int update = 0;
/**
\fn void Timer_ISR()
\brief Timer1 interrupt service routine
\return
*/
void Timer_ISR()
{
if (TIMER_GetInterruptFlags(TIMER_INSTANCE) & TIMER_Match0InterruptFlag)
{
update = 1;
TIMER_ClearInterruptFlags(TIMER_INSTANCE, TIMER_Match0InterruptFlag);
}
}
// Press Button to stop PWM output to check the stop option
static void TIMER_GpioIsr()
{
//Save current irq mask and diable whole port interrupts to get rid of interrupt overflow
uint16_t portIrqMask = GPIO_SaveAndSetIRQMask(BUTTON_GPIO_INSTANCE);
if (GPIO_GetInterruptFlags(BUTTON_GPIO_INSTANCE) & (1 << BUTTON_GPIO_PIN))
{
GPIO_ClearInterruptFlags(BUTTON_GPIO_INSTANCE, 1 << BUTTON_GPIO_PIN);
TIMER_Stop(PWM_INSTANCE);
}
GPIO_RestoreIRQMask(BUTTON_GPIO_INSTANCE, portIrqMask);
}
/*
In this example, we'll use 2 instance of TIMER,
one is used to output pwm, and its frequency is configure as 1KHz while duty cycle
is controlled by another TIMER, the duty cycle is updated in interval of 5 from 0 to 100 periodically,
the period is achieved with aid of another instance of timer whose interrupt is triggered every 500ms.
*/
void TIMER_PWMdemo(void)
{
// pwm output and normal timer interrupt test
unsigned int dutyCycle = PWM_INIT_DUTY_CYCLE_PERCENT;
printf("TIMER PWM Example Start\r\n");
// PWM out pin config
pad_config_t config;
PAD_GetDefaultConfig(&config);
config.mux = PWM_OUT_ALT_SEL;
PAD_SetPinConfig(PWM_OUT_PIN, &config);
config.mux = PWM_OUT_N_ALT_SEL;
PAD_SetPinConfig(PWM_OUT_N_PIN, &config);
// Button pin config
config.mux = BUTTON_PAD_ALT_FUNC;
PAD_SetPinConfig(BUTTON_PAD_INDEX, &config);
PAD_SetPinPullConfig(BUTTON_PAD_INDEX, PAD_InternalPullUp);
gpio_pin_config_t gpioConfig;
gpioConfig.pinDirection = GPIO_DirectionInput;
gpioConfig.misc.interruptConfig = GPIO_InterruptFallingEdge;
GPIO_PinConfig(BUTTON_GPIO_INSTANCE, BUTTON_GPIO_PIN, &gpioConfig);
// Enable IRQ
XIC_SetVector(PXIC_Gpio_IRQn, TIMER_GpioIsr);
XIC_EnableIRQ(PXIC_Gpio_IRQn);
// Config PWM clock
CLOCK_SetClockSrc(PWM_CLOCK_ID, PWM_CLOCK_SOURCE);
CLOCK_SetClockDiv(PWM_CLOCK_ID, 1);
TIMER_DriverInit();
timer_pwm_config_t pwmConfig;
pwmConfig.pwmFreq_HZ = 100;
pwmConfig.srcClock_HZ = GPR_GetClockFreq(PWM_CLOCK_ID);
pwmConfig.stopOption = TIMER_PwmStopLow;
pwmConfig.dutyCyclePercent = PWM_INIT_DUTY_CYCLE_PERCENT;
TIMER_SetupPwm(PWM_INSTANCE, &pwmConfig);
// TIMER config
// Config TIMER clock
CLOCK_SetClockSrc(TIMER_CLOCK_ID, TIMER_CLOCK_SOURCE);
CLOCK_SetClockDiv(TIMER_CLOCK_ID, 1);
// Config TIMER period as 500ms, match0 value is 20000
timer_config_t timerConfig;
TIMER_GetDefaultConfig(&timerConfig);
timerConfig.initValue = 1;
timerConfig.reloadOption = TIMER_ReloadOnMatch0;
timerConfig.match0 = 20000;
TIMER_Init(TIMER_INSTANCE, &timerConfig);
TIMER_InterruptConfig(TIMER_INSTANCE, TIMER_Match0Select, TIMER_InterruptLevel);
TIMER_InterruptConfig(TIMER_INSTANCE, TIMER_Match1Select, TIMER_InterruptDisabled);
TIMER_InterruptConfig(TIMER_INSTANCE, TIMER_Match2Select, TIMER_InterruptDisabled);
// Enable TIMER IRQ
XIC_SetVector(TIMER_INSTANCE_IRQ, Timer_ISR);
XIC_EnableIRQ(TIMER_INSTANCE_IRQ);
TIMER_Start(PWM_INSTANCE);
TIMER_Start(TIMER_INSTANCE);
while(1)
{
if(update == 1)
{
update = 0;
dutyCycle += 5;
if (dutyCycle > 100)
dutyCycle = 0;
TIMER_UpdatePwmDutyCycle(PWM_INSTANCE, dutyCycle);
printf("current dutyCycle: %d\r\n", dutyCycle);
}
}
}
/*
In this example, we'll use 2 instance of TIMER,
one is used to output pwm on 100Hz frequency, the other works as a counter and its input is just from pwm output, we keep sampling the counter value
in interval of 500ms
*/
void TIMER_CounterDemo(void)
{
printf("TIMER Counter Example Start\r\n");
// PWM out pin config
pad_config_t config;
PAD_GetDefaultConfig(&config);
config.mux = PWM_OUT_ALT_SEL;
PAD_SetPinConfig(PWM_OUT_PIN, &config);
config.mux = COUNTER_INPUT_ALT_SEL;
PAD_SetPinConfig(COUNTER_INPUT_PIN, &config);
// Config PWM clock
CLOCK_SetClockSrc(PWM_CLOCK_ID, PWM_CLOCK_SOURCE);
CLOCK_SetClockDiv(PWM_CLOCK_ID, 1);
TIMER_DriverInit();
timer_pwm_config_t pwmConfig;
pwmConfig.pwmFreq_HZ = 100;
pwmConfig.srcClock_HZ = GPR_GetClockFreq(PWM_CLOCK_ID);
pwmConfig.stopOption = TIMER_PwmStopLow;
pwmConfig.dutyCyclePercent = 50;
TIMER_SetupPwm(PWM_INSTANCE, &pwmConfig);
TIMER_Start(PWM_INSTANCE);
// COUNTER config
timer_config_t counterConfig;
TIMER_GetDefaultConfig(&counterConfig);
counterConfig.clockSource= TIMER_ExternalClock;
counterConfig.captureEdge = TIMER_CaptureFallingEdge;
counterConfig.reloadOption = TIMER_ReloadDisabled;
// Connect GPIO2 Pin with PWM output Pin so that we can counter on the number of pwm pulse
TIMER_SetExternalClockInput(COUNTER_INSTANCE, 0, 2, true);
TIMER_Init(COUNTER_INSTANCE, &counterConfig);
TIMER_Start(COUNTER_INSTANCE);
while(1)
{
printf("counter value: %d\r\n", TIMER_GetCount(COUNTER_INSTANCE));
delay_us(500000);
}
}
timer_capture_result_t captureResult = {0};
volatile static int captureDone = 0;
void Capture_ISR()
{
// if pulse type, don't need to check
if (TIMER_GetInterruptFlags(COUNTER_INSTANCE) & TIMER_CpatureInterruptFlag)
{
TIMER_GetCaptureResult(COUNTER_INSTANCE, &captureResult);
captureDone = 1;
TIMER_ClearInterruptFlags(COUNTER_INSTANCE, TIMER_CpatureInterruptFlag);
}
}
/*
In this example, we'll use 2 instance of TIMER,
one is used to output pwm on 100Hz frequency, the other works as a capturer and its input is just from pwm output
*/
void TIMER_CaptureDemo(void)
{
printf("TIMER Capture Example Start\r\n");
// PWM out pin config
pad_config_t config;
PAD_GetDefaultConfig(&config);
config.mux = PWM_OUT_ALT_SEL;
PAD_SetPinConfig(PWM_OUT_PIN, &config);
config.mux = CAPTURE_INPUT_ALT_SEL;
PAD_SetPinConfig(CAPTURE_INPUT_PIN, &config);
// Config PWM clock
CLOCK_SetClockSrc(PWM_CLOCK_ID, PWM_CLOCK_SOURCE);
CLOCK_SetClockDiv(PWM_CLOCK_ID, 1);
// Config CAPTURE clock,
CLOCK_SetClockSrc(CAPTURE_CLOCK_ID, CAPTURE_CLOCK_SOURCE);
CLOCK_SetClockDiv(CAPTURE_CLOCK_ID, 1);
TIMER_DriverInit();
timer_pwm_config_t pwmConfig;
pwmConfig.pwmFreq_HZ = 100;
pwmConfig.srcClock_HZ = GPR_GetClockFreq(PWM_CLOCK_ID);
pwmConfig.stopOption = TIMER_PwmStopHigh;
pwmConfig.dutyCyclePercent = 50;
TIMER_SetupPwm(PWM_INSTANCE, &pwmConfig);
TIMER_Start(PWM_INSTANCE);
// CAPTURE config
timer_config_t captureConfig;
TIMER_GetDefaultConfig(&captureConfig);
captureConfig.clockSource= TIMER_ExternalClock;
captureConfig.captureEdge = TIMER_CaptureBothEdge;
captureConfig.captureMode = TIMER_CaptureModeEnable;
captureConfig.reloadOption = TIMER_ReloadDisabled;
// Connect GPIO2 Pin with PWM output Pin so that we can capture on the pwm pulse
TIMER_SetExternalClockInput(CAPTURE_INSTANCE, 0, 2, true);
TIMER_Init(CAPTURE_INSTANCE, &captureConfig);
TIMER_InterruptConfig(CAPTURE_INSTANCE, TIMER_CaptureInterruptEnable, TIMER_InterruptLevel);
// Enable CAPTURE IRQ
XIC_SetVector(CAPTURE_INSTANCE_IRQ, Capture_ISR);
XIC_EnableIRQ(CAPTURE_INSTANCE_IRQ);
TIMER_Start(CAPTURE_INSTANCE);
while(1)
{
if(captureDone == 1)
{
captureDone = 0;
// the diff of capture value: CAPTURE internal frequecy / pwm frequency
printf("capture value: %d, edge: %d\r\n", captureResult.counterValue, captureResult.capturedEdge);
}
}
}
#define TIMER_FEATURE_COUNTER 1
#define TIMER_FEATURE_CAPTURE 2
#define TIMER_FEATURE_PWM 3
void TIMER_ExampleEntry(void)
{
uint32_t featureSel = TIMER_FEATURE_CAPTURE;
if(featureSel == TIMER_FEATURE_PWM)
{
TIMER_PWMdemo();
}
else if(featureSel == TIMER_FEATURE_CAPTURE)
{
TIMER_CaptureDemo();
}
else if(featureSel == TIMER_FEATURE_COUNTER)
{
TIMER_CounterDemo();
}
while(1);
}
